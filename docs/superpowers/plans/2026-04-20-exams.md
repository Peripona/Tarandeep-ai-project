# Level Exams Sub-project Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add one standalone exam per CEFR level (A1–C2), each divided into Reading, Grammar, and Conversation sections, with a final score screen and a pass/fail badge on the Progress page's CEFR rings.

**Architecture:** Exam content in `src/content/exams.ts`, re-exported via `catalog.ts`. A new `examResults` Zustand slice stores the latest attempt per level. Two new pages (`/exams`, `/exams/[level]`) and two components (`ExamCard`, `ExamView`). Progress page CEFR rings updated to show a pass/fail badge. Exams added to the Sidebar and mobile More drawer.

**Tech Stack:** Next.js 15 App Router, Zustand 5, Vitest 4, Tailwind CSS 4, Radix UI, lucide-react

**Prerequisite:** Reading and Conversation sub-projects must be complete (types, store, `levelAssessor` signature, More drawer all exist).

```typescript
import { describe, it, expect } from "vitest";
import { levelExams } from "@/content/exams";
import type { LevelExam } from "@/lib/types";

describe("exam content", () => {
  it("exports exactly 6 exams", () => {
    expect(levelExams).toHaveLength(6);
  });

  it("one exam per CEFR level", () => {
    const levels = levelExams.map((e) => e.level);
    expect(new Set(levels).size).toBe(6);
    ["A1","A2","B1","B2","C1","C2"].forEach((l) => {
      expect(levels).toContain(l);
    });
  });

  it("each exam has required fields", () => {
    levelExams.forEach((e: LevelExam) => {
      expect(e.level).toBeTruthy();
      expect(e.title).toBeTruthy();
      expect(e.description).toBeTruthy();
      expect(e.sections.length).toBeGreaterThanOrEqual(2);
    });
  });

  it("each section has a title and questions", () => {
    levelExams.forEach((e) => {
      e.sections.forEach((s) => {
        expect(s.title).toBeTruthy();
        expect(s.questions.length).toBeGreaterThanOrEqual(2);
      });
    });
  });

  it("all question ids are unique within an exam", () => {
    levelExams.forEach((e) => {
      const ids = e.sections.flatMap((s) => s.questions.map((q) => q.id));
      expect(new Set(ids).size).toBe(ids.length);
    });
  });
});
```

## Task 6: Build ExamView component

**Files:**
- Create: `src/components/exams/ExamView.tsx`

- [ ] **Step 1: Create the component**

```typescript
"use client";

import { useState, useCallback } from "react";
import { useRouter } from "next/navigation";
import type { LevelExam } from "@/lib/types";
import { useAppStore } from "@/lib/store";
import { FillBlank } from "@/components/grammar/FillBlank";
import { MultipleChoice } from "@/components/grammar/MultipleChoice";
import { Button } from "@/components/ui/button";
import { LevelBadge } from "@/components/ui/LevelBadge";

export function ExamView({ exam }: { exam: LevelExam }) {
  const router = useRouter();
  const recordExamResult = useAppStore((s) => s.recordExamResult);
  const [sectionIndex, setSectionIndex] = useState(0);
  const [scores, setScores] = useState<Record<string, boolean>>({});
  const [submitted, setSubmitted] = useState(false);

  const currentSection = exam.sections[sectionIndex];
  const totalQuestions = exam.sections.reduce((sum, s) => sum + s.questions.length, 0);
  const answeredInSection = currentSection.questions.filter((q) => scores[q.id] !== undefined).length;
  const sectionComplete = answeredInSection === currentSection.questions.length;

  const onAnswer = useCallback((id: string, correct: boolean) => {
    setScores((s) => ({ ...s, [id]: correct }));
  }, []);

  const goNextSection = () => {
    if (sectionIndex < exam.sections.length - 1) {
      setSectionIndex((i) => i + 1);
    } else {
      setSubmitted(true);
      const correct = Object.values(scores).filter(Boolean).length;
      recordExamResult(exam.level, correct, totalQuestions);
    }
  };

  const correct = Object.values(scores).filter(Boolean).length;
  const pct = submitted ? correct / totalQuestions : 0;
  const passed = pct >= 0.8;

  if (submitted) {
    return (
      <div className="mx-auto max-w-lg space-y-6 text-center py-16">
        <LevelBadge level={exam.level} />
        <h1 className="text-3xl font-bold">{passed ? "Passed!" : "Not passed"}</h1>
        <p className="text-xl">
          {correct} / {totalQuestions} ({Math.round(pct * 100)}%)
        </p>
        <p className="text-muted-foreground">
          {passed
            ? `You have demonstrated ${exam.level} level proficiency.`
            : `You need 80% to pass. Keep studying and try again!`}
        </p>
        <div className="space-y-2">
          {exam.sections.map((s) => {
            const sCorrect = s.questions.filter((q) => scores[q.id]).length;
            return (
              <div key={s.title} className="flex justify-between rounded-lg border border-border px-4 py-2 text-sm">
                <span>{s.title}</span>
                <span>{sCorrect}/{s.questions.length}</span>
              </div>
            );
          })}
        </div>
        <Button onClick={() => router.push("/exams")}>Back to Exams</Button>
      </div>
    );
  }

  return (
    <div className="mx-auto max-w-2xl space-y-8">
      {/* Header */}
      <div className="space-y-1">
        <div className="flex items-center gap-2">
          <LevelBadge level={exam.level} />
          <span className="text-sm text-muted-foreground">
            Section {sectionIndex + 1} of {exam.sections.length}
          </span>
        </div>
        <h1 className="text-2xl font-semibold">{exam.title}</h1>
        <p className="text-sm text-muted-foreground">{exam.description}</p>
      </div>

      {/* Section progress bar */}
      <div className="flex gap-1.5">
        {exam.sections.map((s, i) => (
          <div
            key={s.title}
            className={`h-1.5 flex-1 rounded-full ${
              i < sectionIndex
                ? "bg-primary"
                : i === sectionIndex
                ? "bg-primary/50"
                : "bg-muted"
            }`}
          />
        ))}
      </div>

      {/* Section */}
      <div className="space-y-4">
        <h2 className="text-lg font-semibold">{currentSection.title}</h2>
        {currentSection.questions.map((q) => {
          if (q.type === "multipleChoice") {
            return <MultipleChoice key={q.id} exercise={q} onAnswer={(c) => onAnswer(q.id, c)} />;
          }
          return <FillBlank key={q.id} exercise={q} onAnswer={(c) => onAnswer(q.id, c)} />;
        })}
      </div>

      <Button
        disabled={!sectionComplete}
        onClick={goNextSection}
        className="w-full"
      >
        {sectionIndex < exam.sections.length - 1 ? "Next Section →" : "Submit Exam"}
      </Button>
    </div>
  );
}
```
